## 3.1 RECURSIVE SELF-ATTENTION NETWORK (RSAN) SPECIFICATION

### Architecture Definition:

```python
class RSANLayer:
    """
    Single layer of Recursive Self-Attention Network
    Implements one level of Theory of Mind reasoning
    """
    def __init__(self, n_agents, d_model, n_heads):
        self.n_agents = n_agents  # Number of agents being modeled
        self.d_model = d_model    # Dimension of model
        self.n_heads = n_heads    # Attention heads per agent
        
        # Per-agent attention heads
        self.agent_attention = [
            MultiHeadAttention(d_model, n_heads) 
            for _ in range(n_agents)
        ]
        
        # Meta-attention (attends to other agents' representations)
        self.meta_attention = MultiHeadAttention(d_model, n_heads)
        
        # Skip connection for direct perception
        self.skip_gate = LearnedGate(d_model)
        
    def forward(self, observations, agent_states):
        """
        observations: Direct perceptual input
        agent_states: Previous mental state representations
        """
        # Per-agent mental state modeling
        agent_models = []
        for i, attn in enumerate(self.agent_attention):
            # Attend to observations relevant to agent i
            agent_model = attn(
                query=agent_states[i],
                key=observations,
                value=observations
            )
            agent_models.append(agent_model)
        
        # Meta-level: Attend to other agents' models
        stacked_models = stack(agent_models)
        meta_representation = self.meta_attention(
            query=stacked_models,
            key=stacked_models,
            value=stacked_models
        )
        
        # Skip connection: Combine with direct observation
        gated_skip = self.skip_gate(observations)
        output = meta_representation + gated_skip
        
        return output, agent_models
```

### Stacking for ToM Depth:

```python
class RecursiveToMNetwork:
    """
    Full recursive Theory of Mind network
    Depth corresponds to ToM orders (1-5)
    """
    def __init__(self, tom_depth, n_agents, d_model):
        self.tom_depth = tom_depth
        self.layers = [
            RSANLayer(n_agents, d_model, n_heads=4)
            for _ in range(tom_depth)
        ]
        
    def forward(self, observations, initial_states):
        """
        Compute nested mental state models
        Layer 1: What do they want/believe? (1st order)
        Layer 2: What do they think I want? (2nd order)
        Layer 3: What do they think I think they want? (3rd order)
        etc.
        """
        current_states = initial_states
        all_agent_models = []
        
        for layer in self.layers:
            output, agent_models = layer(observations, current_states)
            all_agent_models.append(agent_models)
            current_states = output
            
        return current_states, all_agent_models
```

### NAS Search Space:

The Neural Architecture Search explores:
- Number of attention heads per agent (2-8)
- Skip connection patterns (none, residual, gated)
- Layer normalization placement
- Depth-wise vs. width-wise scaling
- Meta-attention vs. cross-attention variants

### Discovered Optimal Features:
1. **Gated skip connections** essential for social scenarios
2. **Separate goal/belief/emotion heads** outperform unified heads
3. **Meta-attention should attend to fewer agents than base attention**
4. **Depth 5 shows diminishing returns** (validates human ToM limits)

    Single unit of Transparent Recurrent Network
    Each unit corresponds to one Soul Map dimension
    """
    def __init__(self, dimension_name, d_state):
        self.dimension_name = dimension_name
        self.d_state = d_state
        
        # Gates with interpretable names
        self.update_gate = nn.Linear(d_state * 2, d_state)  # New info integration
        self.reset_gate = nn.Linear(d_state * 2, d_state)   # Contradiction handling
        self.output_gate = nn.Linear(d_state * 2, d_state)  # Behavior selection
        
        # Candidate update computation
        self.candidate = nn.Linear(d_state * 2, d_state)
        
    def forward(self, input_t, state_t_minus_1):
        """
        input_t: Current input (observations, other agents' states)
        state_t_minus_1: Previous state of this dimension
        """
        combined = concat(input_t, state_t_minus_1)
        
        # Compute gates (all interpretable)
        update = sigmoid(self.update_gate(combined))
        reset = sigmoid(self.reset_gate(combined))
        
        # Compute candidate new state
        reset_state = reset * state_t_minus_1
        candidate_combined = concat(input_t, reset_state)
        candidate = tanh(self.candidate(candidate_combined))
        
        # Update state
        new_state = (1 - update) * state_t_minus_1 + update * candidate
        
        # Output gate for behavior
        output = sigmoid(self.output_gate(concat(input_t, new_state)))
        behavior_contribution = output * new_state
        
        # Return new state AND interpretable components
        return new_state, {
            'update_strength': update,
            'reset_strength': reset,
            'output_strength': output,
            'state_delta': new_state - state_t_minus_1
        }
```

class SoulMap:
    """
    Complete Soul Map as 60 Transparent Recurrent Units
    """
    def __init__(self, dimension_definitions):
        self.units = {}
        for dim_def in dimension_definitions:
            self.units[dim_def.name] = TransparentRecurrentUnit(
                dimension_name=dim_def.name,
                d_state=dim_def.state_size
            )
        
        # Cross-dimension attention (dimensions affect each other)
        self.cross_attention = CrossDimensionAttention(
            n_dimensions=len(dimension_definitions)
        )
        
    def update(self, observations, time_delta):
        """
        Update all Soul Map dimensions given new observations
        """
        new_states = {}
        interpretable_updates = {}
        
        # First pass: Individual unit updates
        for name, unit in self.units.items():
            relevant_obs = self.extract_relevant(observations, name)
            new_state, update_info = unit.forward(
                relevant_obs, 
                self.current_states[name]
            )
            new_states[name] = new_state
            interpretable_updates[name] = update_info
        
        # Second pass: Cross-dimension interactions
        cross_effects = self.cross_attention(new_states)
        
        # Apply cross-effects
        for name in new_states:
            new_states[name] += cross_effects[name]
        
        self.current_states = new_states
        return interpretable_updates  # Player can see these!
```

class NPCMind:
    """
    Complete NPC cognitive architecture
    Integrates RSAN (ToM reasoning) with TRN (Soul Map persistence)
    """
    def __init__(self, npc_definition):
        # Initialize Soul Map from definition
        self.soul_map = SoulMap(DIMENSION_DEFINITIONS)
        self.soul_map.initialize_from(npc_definition.initial_values)
        
        # Initialize ToM network based on ToM Depth dimension
        tom_depth = int(self.soul_map.get('Theory of Mind Depth'))
        self.tom_network = RecursiveToMNetwork(
            tom_depth=tom_depth,
            n_agents=MAX_TRACKED_AGENTS,
            d_model=256
        )
        
        # Memory systems
        self.episodic_memory = EpisodicMemory(capacity=1000)
        self.relationship_memory = RelationshipMemory()
        self.goal_stack = GoalStack()
        
        # Action selection
        self.action_selector = ActionSelector(self.soul_map)
        
    def perceive(self, world_state):
        """Process environmental input"""
        # Extract relevant observations
        observations = self.perception_module(world_state)
        
        # Update Soul Map based on observations
        soul_updates = self.soul_map.update(observations, world_state.time_delta)
        
        return observations, soul_updates
        
    def reason_about_others(self, observations, other_agents):
        """Apply Theory of Mind to other agents"""
        # Get current mental models of others
        current_models = [
            self.relationship_memory.get_model(agent)
            for agent in other_agents
        ]
        
        # Apply recursive reasoning
        updated_models, model_hierarchy = self.tom_network(
            observations, 
            current_models
        )
        
        # Store updated models
        for agent, model in zip(other_agents, updated_models):
            self.relationship_memory.update_model(agent, model)
        
        return model_hierarchy  # All levels of ToM reasoning
        
    def select_action(self, observations, tom_results):
        """Choose behavior based on goals, Soul Map, and ToM"""
        # Get current goals
        active_goals = self.goal_stack.get_active()
        
        # Generate action candidates
        candidates = self.action_selector.generate_candidates(
            observations=observations,
            soul_map=self.soul_map,
            goals=active_goals,
            tom_models=tom_results
        )
        
        # Evaluate candidates
        scored_candidates = []
        for candidate in candidates:
            score = self.action_selector.evaluate(
                candidate,
                soul_map=self.soul_map,
                goals=active_goals,
                tom_predictions=self.predict_outcomes(candidate, tom_results)
            )
            scored_candidates.append((candidate, score))
        
        # Select best action
        best_action = max(scored_candidates, key=lambda x: x[1])
        
        return best_action
        
    def tick(self, world_state, other_agents):
        """Main update loop - called each game tick"""
        # Perceive
        observations, soul_updates = self.perceive(world_state)
        
        # Reason about others
        tom_results = self.reason_about_others(observations, other_agents)
        
        # Select action
        action, confidence = self.select_action(observations, tom_results)
        
        # Update memories
        self.episodic_memory.store(observations, action)
        
        # Return action and interpretable state (for player visibility)
        return action, {
            'soul_updates': soul_updates,
            'tom_models': tom_results,
            'goals': self.goal_stack.get_active(),
            'confidence': confidence
        }
